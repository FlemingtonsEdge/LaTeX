	\subsection{Aligned}
		\begin{equation*}
			\left\lbrace
			\begin{aligned}
				x+y&=1\\
				x-y&=1
			\end{aligned}
			\right.
			\Longrightarrow
			\left\lbrace
			\begin{aligned}
				x&=1\\
				y&=0
			\end{aligned}
			\right.
		\end{equation*}
